\begin{vdm_al}
class Timer 

instance variables
    public currentTime : nat := 0;
    private static timerInstance : Timer := new Timer();

values 
    stepLength : nat = 100;

operations

    -- Class Constructor
    private Timer: () ==> Timer
    Timer() ==
        skip;

    pure public static GetInstance: () ==> Timer
    GetInstance() ==
        return timerInstance;

    public StepTime: () ==> ()
    StepTime() == 
     currentTime := currentTime + stepLength;
    
    pure public GetTime : () ==> nat
    GetTime() ==
        return currentTime;

end Timer
